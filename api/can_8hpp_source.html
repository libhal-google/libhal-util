57[[nodiscard]] constexpr std::optional<std::uint32_t> is_valid(
58 const can::settings& p_settings,
59 hertz p_operating_frequency) noexcept
60{
61 // 1. propagation_delay, phase_segment1, phase_segment2 and
62 // synchronization_jump_width must be nonzero.
63 if (p_settings.propagation_delay == 0 || p_settings.phase_segment1 == 0 ||
64 p_settings.phase_segment2 == 0 ||
65 p_settings.synchronization_jump_width == 0) {
66 return std::nullopt;
67 }
68
69 // 2. synchronization_jump_width must be the lesser between phase_segment1
70 // and phase_segment2.
71 if (p_settings.synchronization_jump_width > 4 ||
72 p_settings.synchronization_jump_width > p_settings.phase_segment1 ||
73 p_settings.synchronization_jump_width > p_settings.phase_segment2) {
74 return std::nullopt;
75 }
76
77 const std::uint16_t bit_width_v = bit_width(p_settings);
78
79 // 3. The total bit width must be equal to or greater than 8 Tq/bit; the
80 // sum of sync_segment, propagation_delay, phase_segment1 and
81 // phase_segment2.
82 if (bit_width_v < 8) {
83 return std::nullopt;
84 }
85
86 // 4. The CAN device's operating frequency must be at least 8 times the
87 // bit rate to give the minimum.
88 // 5. The ratio between the CAN device's operating frequency and the bit
89 // width must be close enough to an integer to produce a usable BRP.
90 const float bit_width_float = bit_width_v;
91 const float scaled_baud = p_settings.baud_rate * bit_width_float;
92 const float baud_rate_prescaler = p_operating_frequency / scaled_baud;
93 const auto final_prescaler = std::lround(baud_rate_prescaler);
94
95 if (final_prescaler == 0) {
96 return std::nullopt;
97 }
98
99 return final_prescaler;
100}
