56 template<std::unsigned_integral T>
57 constexpr auto origin_mask() const
58 {
59 // At compile time, generate variable containing all 1s with the size of the
60 // target parameter.
61 constexpr T field_of_ones = std::numeric_limits<T>::max();
62
63 // At compile time calculate the number of bits in the target parameter.
64 constexpr size_t target_width = sizeof(T) * 8;
65
66 // Create mask by shifting the set of 1s down so that the number of 1s from
67 // bit position 0 is equal to the width parameter.
