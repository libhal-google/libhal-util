23[[nodiscard]] constexpr std::int64_t cycles_per(
24 hertz p_source,
25 hal::time_duration p_duration) noexcept
26{
27 // Full Equation:
28 // / ratio_num \_
29 // frequency_hz * |period| * | ----------- | = cycles
30 // \ ratio_den /
31 //
32 // std::chrono::nanoseconds::period::num == 1
33 // std::chrono::nanoseconds::period::den == 1,000,000
34
35 const auto denominator = decltype(p_duration)::period::den;
36 const auto float_count = static_cast<float>(p_duration.count());
37 const auto cycle_count = (float_count * p_source) / denominator;
38
39 return static_cast<std::int64_t>(cycle_count);
40}

50template<typename Period>
51constexpr std::chrono::duration<int64_t, Period> wavelength(hertz p_source)
52{
53 using duration = std::chrono::duration<int64_t, Period>;
54
55 static_assert(Period::num == 1, "The period ratio numerator must be 1");
56 static_assert(Period::den >= 1,
57 "The period ratio denominator must be 1 or greater than 1.");
58
59 constexpr auto denominator = static_cast<decltype(p_source)>(Period::den);
60 auto period = (1.0f / p_source) * denominator;
61
62 if (std::isinf(period)) {
63 return duration(std::numeric_limits<int64_t>::max());
64 }
65
66 return duration(static_cast<int64_t>(period));
67}

96[[nodiscard]] inline result<std::chrono::nanoseconds> duration_from_cycles(
97 hertz p_source,
98 uint32_t p_cycles) noexcept
99{
100 // Full Equation (based on the equation in cycles_per()):
101 //
102 //
103 // / cycles * ratio_den \_
104 // |period| = | ---------------------------|
105 // \ frequency_hz * ratio_num /
106 //
107 constexpr auto ratio_den = std::chrono::nanoseconds::period::den;
108 constexpr auto ratio_num = std::chrono::nanoseconds::period::num;
109 constexpr auto int_min = std::numeric_limits<std::int64_t>::min();
110 constexpr auto int_max = std::numeric_limits<std::int64_t>::max();
111 constexpr auto float_int_min = static_cast<float>(int_min);
112 constexpr auto float_int_max = static_cast<float>(int_max);
113
114 const auto source = std::abs(p_source);
115 const auto float_cycles = static_cast<float>(p_cycles);
116 const auto nanoseconds = (float_cycles * ratio_den) / (source * ratio_num);
117
118 if (float_int_min <= nanoseconds && nanoseconds <= float_int_max) {
119 return std::chrono::nanoseconds(static_cast<std::int64_t>(nanoseconds));
120 }
121 return new_error(std::errc::result_out_of_range);
122}
