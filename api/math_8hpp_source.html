84template<typename T>
85[[nodiscard]] constexpr T rounding_division(T p_numerator,
86 T p_denominator) noexcept
87{
88 bool num_sign = p_numerator >= 0;
89 bool den_sign = p_denominator >= 0;
90
91 auto numerator = absolute_value(p_numerator);
92 auto denominator = absolute_value(p_denominator);
93
94 if (denominator > numerator || denominator == 0) {
95 return T{ 0 };
96 }
97
98 const T remainder = numerator % denominator;
99 const T half_denominator = (denominator / 2);
100 T quotient = numerator / denominator;
101 // Round if remainder is greater than half of the denominator
102 if (half_denominator != 0 && remainder >= half_denominator) {
103 quotient++;
104 }
105
106 if (num_sign != den_sign) {
107 return -quotient;
108 }
109
110 return quotient;
111}

145template<std::integral T>
146[[nodiscard]] constexpr std::make_unsigned_t<T> distance(T p_left,
147 T p_right) noexcept
148{
149 static_assert(sizeof(T) <= sizeof(int32_t));
150
151 using unsigned_t = std::make_unsigned_t<T>;
152 // Put left and right values into 64-bit containers to prevent overflow
153 int64_t left{ p_left };
154 int64_t right{ p_right };
155
156 if (right > left) {
157 // Subtraction operation on right to left in this order can never overflow
158 // because the maximum resultant of left and right being INT32_MAX and
159 // INT32_MIN, will equal UINT32_MAX which can be stored within an int64_t
160 // value.
161 int64_t difference = right - left;
162 // Casting this value to the unsigned variant will always fit as the
163 // distance between any 32-bit signed numbers can always fit in a 32-bit
164 // unsigned number.
165 return static_cast<unsigned_t>(difference);
166 } else {
167 // Same logic as the if statement block above.
168 int64_t difference = left - right;
169 return static_cast<unsigned_t>(difference);

181constexpr static bool equals(std::floating_point auto p_value1,
182 std::floating_point auto p_value2,
183 float p_epsilon = 1e-9f)
184{
185 if (p_value1 == p_value2) {
186 return true;
187 }
188 auto value1_abs = std::abs(p_value1);
189 auto value2_abs = std::abs(p_value2);
190 auto diff = std::abs(p_value1 - p_value2);
191 auto absolute_values_sum = value1_abs + value2_abs;
192
193 using float_t = decltype(absolute_values_sum);
194
195 if (p_value1 == 0 || p_value2 == 0 ||
196 (absolute_values_sum < std::numeric_limits<float_t>::min())) {
197 return diff < (p_epsilon * std::numeric_limits<float_t>::min());
198 } else {
199 auto relative_error =
200 diff / std::min(absolute_values_sum, std::numeric_limits<float_t>::max());
201 return relative_error < p_epsilon;
202 }
